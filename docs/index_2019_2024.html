1 Análisis de Lactancia Materna en Chile 2023-2024

1.1 Introducción

Este análisis examina los patrones de lactancia materna en Chile durante los años 2023 y 2024, considerando diferentes tipos de alimentación infantil, meses de vida del lactante y características demográficas.

# ---- 5) Helpers para mapear columnas por año (robusto entre 2019–2024) -------
first_existing <- function(candidates, available) {
  x <- candidates[candidates %in% available]
  if (length(x) == 0) NA_character_ else x[1]
}

get_column_map <- function(dt, ano_chr) {
  ano  <- as.integer(ano_chr)
  cols <- names(dt)

  if (ano <= 2023) {
    # Formato tipo 2023
    tot <- first_existing(c("Col01"), cols)          # TOTAL
    m1  <- first_existing(c("Col03","Col04"), cols)  # 1° mes
    m3  <- first_existing(c("Col04","Col05"), cols)  # 3° mes
    m6  <- first_existing(c("Col05","Col06"), cols)  # 6° mes
    po  <- first_existing(c("Col08","Col09"), cols)  # Pueblos Originarios
    mi  <- first_existing(c("Col09","Col10"), cols)  # Migrantes
  } else {
    # Formato tipo 2024
    tot <- first_existing(c("Col01"), cols)
    m1  <- first_existing(c("Col04","Col03"), cols)
    m3  <- first_existing(c("Col05","Col04"), cols)
    m6  <- first_existing(c("Col06","Col05"), cols)
    po  <- first_existing(c("Col09","Col08"), cols)
    mi  <- first_existing(c("Col10","Col09"), cols)
  }

  columnas <- c(tot, m1, m3, m6, po, mi)
  nombres  <- c("TOTAL","1° mes","3° mes","6° mes","Pueblos Originarios","Migrantes")
  valid    <- !is.na(columnas)

  list(columnas = columnas[valid], nombres = nombres[valid])
}

# ---- 6) Funciones de procesamiento (por año y por comuna) --------------------
procesar_datos_año <- function(dt, ano_chr) {
  map <- get_column_map(dt, ano_chr)

  # Convertir a numérico de forma segura
  dt[, (map$columnas) := lapply(.SD, function(x) as.numeric(gsub(",", ".", as.character(x)))),
     .SDcols = map$columnas]

  # Agregar por código de prestación
  out <- dt[, lapply(.SD, sum, na.rm = TRUE),
            by = .(CodigoPrestacion), .SDcols = map$columnas]
  data.table::setnames(out, map$columnas, map$nombres)

  # Descripciones legibles
  out[, Tipo_Alimentacion := dplyr::recode(
    CodigoPrestacion,
    "A0200002" = "Lactancia Materna Exclusiva",
    "03500359" = "Lactancia Materna / Fórmula Láctea",
    "03500360" = "Fórmula Láctea",
    "A0200003" = "Lactancia Materna + Alimentación Complementaria",
    "03600140" = "LM / Fórmula / Alimentación Complementaria",
    "03600150" = "Fórmula + Alimentación Complementaria",
    "A0200001" = "Total Niños/as Controlados",
    .default     = "Otro"
  )]

  out[, Año := as.integer(ano_chr)]
  data.table::setcolorder(out, c("Año","CodigoPrestacion","Tipo_Alimentacion", map$nombres))
  out[]
}

procesar_datos_comuna <- function(dt, ano_chr) {
  map <- get_column_map(dt, ano_chr)

  dt[, (map$columnas) := lapply(.SD, function(x) as.numeric(gsub(",", ".", as.character(x)))),
     .SDcols = map$columnas]

  out <- dt[, lapply(.SD, sum, na.rm = TRUE),
            by = .(CodigoPrestacion, IdComuna), .SDcols = map$columnas]
  data.table::setnames(out, map$columnas, map$nombres)

  out[, Tipo_Alimentacion := dplyr::recode(
    CodigoPrestacion,

}

library(data.table); library(dplyr); library(tidyr)

build_long_mes <- function(dt) {
  cols_meses <- c("1° mes","3° mes","6° mes")
  use_cols   <- intersect(cols_meses, names(dt))
  if (length(use_cols) == 0) return(NULL)

  d <- data.table::copy(dt)
  # Excluir el denominador de niños controlados si existiera en las columnas de mes
  d <- d[Tipo_Alimentacion != "Total Niños/as Controlados"]

  long <- data.table::melt(
    d,
    id.vars = c("Año","CodigoPrestacion","Tipo_Alimentacion"),
    measure.vars = use_cols,
    variable.name = "Mes",
    value.name = "TotalMes",
    variable.factor = FALSE
  )
  long[, TotalMes := as.numeric(TotalMes)]
  long
}

meses_long_list <- lapply(resumen_anual, build_long_mes)
meses_long <- data.table::rbindlist(
  Filter(Negate(is.null), meses_long_list), use.names = TRUE, fill = TRUE
)

# Denominador por Año-Mes = suma de todos los tipos
den_mes <- meses_long[, .(DenomMes = sum(TotalMes, na.rm = TRUE)), by = .(Año, Mes)]

# Tabla LONG final con % por mes
tabla_meses_long <- den_mes[meses_long, on = c("Año","Mes")][
  , `Porcentaje (%)` := ifelse(DenomMes > 0, round(100 * TotalMes / DenomMes, 2), NA_real_)
][
  , .(Año,
      Mes = factor(Mes, levels = c("1° mes","3° mes","6° mes")),
      Tipo_Alimentacion,
      Total = TotalMes,

][order(Mes, Año, -`Porcentaje (%)`)]

# --- Construir tabla anual: Tipo de lactancia × Grupo poblacional × Año -------
# Requiere: 'resumen_anual' creado antes (con columnas: TOTAL, Pueblos Originarios, Migrantes)

anual_all <- data.table::rbindlist(resumen_anual, use.names = TRUE, fill = TRUE)

stopifnot("Año" %in% names(anual_all), "TOTAL" %in% names(anual_all))

# Usar solo tipos de lactancia (excluir total de niños controlados)
lact <- anual_all[
  Tipo_Alimentacion != "Total Niños/as Controlados" & !is.na(TOTAL)
]

# Asegurar columnas de grupos; si no existen, poner 0
if (!("Pueblos Originarios" %in% names(lact))) lact[, `Pueblos Originarios` := 0]
if (!("Migrantes" %in% names(lact)))          lact[, Migrantes := 0]

# Convertir a numérico de forma segura
to_num <- function(x) as.numeric(gsub(",", ".", as.character(x)))
num_cols <- intersect(c("TOTAL", "Pueblos Originarios", "Migrantes"), names(lact))
lact[, (num_cols) := lapply(.SD, to_num), .SDcols = num_cols]

# Calcular Población General = TOTAL - (PO + Migrantes), truncando en 0
lact[, `Población General` := pmax(
  0,
  TOTAL - (ifelse(is.na(`Pueblos Originarios`), 0, `Pueblos Originarios`) +
           ifelse(is.na(Migrantes), 0, Migrantes))
)]

# Pasar a formato largo por grupo
long_grupo <- data.table::melt(
  lact,
  id.vars = c("Año","CodigoPrestacion","Tipo_Alimentacion"),
  measure.vars = c("Pueblos Originarios","Migrantes","Población General"),
  variable.name = "Grupo", value.name = "TotalGrupo",
  variable.factor = FALSE
)

# Denominador por (Año, Grupo): suma de todos los tipos
den_grupo <- long_grupo[, .(DenomGrupo = sum(TotalGrupo, na.rm = TRUE)), by = .(Año, Grupo)]

# Tabla final con % por (Año, Grupo, Tipo_Alimentacion)
tabla_grupo_lact_long <- den_grupo[long_grupo, on = c("Año","Grupo")][
  , `Porcentaje (%)` := ifelse(DenomGrupo > 0, round(100 * TotalGrupo / DenomGrupo, 2), NA_real_)
][
  , .(Año,
      Grupo = factor(Grupo, levels = c("Población General","Pueblos Originarios","Migrantes")),
      Tipo_Alimentacion,
      Total = TotalGrupo,
      `Porcentaje (%)`)
][order(Grupo, Año, -`Porcentaje (%)`)]
